from functools import reduce

def circumferenceOfCircle():
    radius = float(input("Enter a radius"))
    print("Circumference is", 2*radius*3.1415926)

def areaOfCircle():
    radius = float(input("Enter a radius"))
    print("Area is", radius**2 * 3.1415926)

def costOfPizza():
    dia = float(input("Enter a diameter"))
    print("your pizza costs", 1.5*3.1415926*(dia/2)**2)

def slopeOfLine():
    x1 = float(input("Enter x of point 1"))
    y1 = float(input("Enter y of point 1"))
    x2 = float(input("Enter x of point 2"))
    y2 = float(input("Enter y of point 2"))
    print("Your gradient is", (y2 - y1) / (x2 - x1))

def distanceBetweenPoints():
    x1 = float(input("Enter x of point 1"))
    y1 = float(input("Enter y of point 1"))
    x2 = float(input("Enter x of point 2"))
    y2 = float(input("Enter y of point 2"))

    print("Length is", ((x2-x1)**2 + (y2-y1)**2) ** 0.5)

def travelStatistics():
    avgSpeed = float(input("Enter the average speed of the car "))
    duration = float(input("Enter the duration "))

    print("fuel used was ", avgSpeed * duration * 5)

def averageOfNumbers():
    n = int(input("How many numbers "))

    total = 0

    for _ in range(n):
        total += float(input("Enter a number "))

    print("Your average is ", total / n)

def selectCoins():
    amount = round(float(input("Enter your amount of money")), 2)

    coins = {2: 0, 1: 0, 0.5: 0, 0.2: 0, 0.1: 0, 0.05: 0, 0.02: 0, 0.01: 0}

    for k, v in coins:
        while amount >= k:
            v += 1
            amount -= k
        print(k, ":", v)
