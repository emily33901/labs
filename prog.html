def selectCoins():
    amount = round(float(input("Enter your amount of money")), 2)

    coins = {2: 0, 1: 0, 0.5: 0, 0.2: 0, 0.1: 0, 0.05: 0, 0.02: 0, 0.01: 0}

    for k, v in coins:
        while amount >= k:
            v += 1
            amount -= k
        print(k, ":", v)

from functools import reduce
        
def displayTextWithSpaces(win, pos, text, pointSize):
    t = Text(pos, reduce(lambda x, y: x + " " + y.upper(), text, ""))
    t.setFace('times roman')
    t.setSize(pointSize)
    t.draw(win)

def drawStickFigure(win, pos, size):
    head = Circle(Point(pos.x, pos.y - size - 20), 20)
    head.draw(win)
    body = Line(Point(pos.x, pos.y - size), pos)
    body.draw(win)
    arm1 = Line(Point(pos.x, pos.y - size + 10), Point(pos.x+20, pos.y-20))
    arm1.draw(win)
    arm2 = Line(Point(pos.x, pos.y - size + 10), Point(pos.x-20, pos.y-20))
    arm2.draw(win)
    
    leg1 = Line(pos, Point(pos.x + 20, pos.y + 20))
    leg1.draw(win)
    leg2 = Line(pos, Point(pos.x - 20, pos.y + 20))
    leg2.draw(win)
